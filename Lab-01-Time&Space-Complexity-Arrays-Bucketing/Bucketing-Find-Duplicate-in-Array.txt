Find Duplicate in Array (Bucketing Algorithm)

Given a read only array of n + 1 integers between 1 and n, find one number that repeats.

INPUT(s):

11

10 9 7 6 5 1 2 3 8 4 7

OUTPUT(s):

7

INPUT(s):

5

1 2 3 4 4

OUTPUT(s):

4

INPUT(s):

5

1 1 2 3 4

OUTPUT(s):

1

#include <stdio.h>
#include <string.h>
int main(){
    int n;
    scanf("%d",&n);
    int arr[n],buc[n];
    memset(buc,0,sizeof(buc));
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
        buc[arr[i]]++;
    }
    for(int i=0;i<n;i++){
        if(buc[i]>1){
            printf("%d",i);
            return 0;
        }
    }
}
